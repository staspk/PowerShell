wsl -d {name}			=>	Launch distro with it's {name}
wsl -l/--list			=>	List distro's {name}
exit					=>	Logout of the distro back into Windows Terminal
wsl --shutdown			=>  Shut down all distros

wsl -l -v/--verbose		=> "Have all distros been shut down?"


HOW TO INSTALL A NEW DISTRO {
	wsl --list --online												=>	See list of distros available
	wsl --install -d Ubuntu-24.04 --name Ubuntu-24-{project}		=>	Install the desired distro
}
